                    //Loop through the bandates array and add the dates to the dropdown for the current banlisttype
                    jQuery.each(bandates, function (i, item) {
                        if (item.type == banlisttype) {

                            //Get Text for the Date in the format of Month Year
                            var date = new Date(item.date);
                            var month = date.toLocaleString('default', {month: 'long'});
                            var year = date.getFullYear();

                            jQuery("#banlistdate").append("<option value='" + item.date + "'>" + month + " " + year + "</option>");
                        }
                    });

                    //Check if query string for date is set
                    if (getQueryVariable("date") != null) {
                        //If query string is set, set banlistdate to the query string
                        var banlistdate = decodeURIComponent(getQueryVariable("date"));

                        //Change the selected option in the dropdown to the banlistdate
                        jQuery("#banlistdate").val(banlistdate);
                    }

                    //Get the banlist
                    updateList(banlisttype);
                },
                error: function () {
                    //If the banlist dates are not valid, display an error
                    jQuery("#banlistdate").html("<option value=''>There was an error getting the banlist dates. Please try again later.</option>");
                }
            });

        });

        //When banlisttype is changed, update the banlist
        jQuery("#banlisttype").on("change", function () {

            //Set banlisttype to the selected value
            banlisttype = jQuery("#banlisttype").val();

            //Upate the dates in the dropdown
            jQuery("#banlistdate").html("");
            jQuery.each(bandates, function (i, item) {
                if (item.type == banlisttype) {

                    //Get Text for the Date in the format of Month Year
                    var date = new Date(item.date);
                    var month = date.toLocaleString('default', {month: 'long'});
                    var year = date.getFullYear();

                    jQuery("#banlistdate").append("<option value='" + item.date + "'>" + month + " " + year + "</option>");
                }
            });

            updateList(banlisttype);
        });

        //When banlistdate is changed
        jQuery("#banlistdate").on("change", function () {
            updateList(banlisttype);
        });

        //Query String
        function getQueryVariable(variable) {
            var query = window.location.search.substring(1);
            var vars = query.split("&");
            for (var i = 0; i < vars.length; i++) {
                var pair = vars[i].split("=");
                if (pair[0] == variable) {
                    return pair[1];
                }
            }
            return (null);
        }

        //Get the banlist
        function updateList(banlisttype) {

            //Get Date
            var banlistdate = jQuery("#banlistdate").val();

            //Ajax call to get the banlist
            jQuery.ajax({
                url: "/api/banlist/getBanList.php?list=" + banlisttype+"&date="+banlistdate,
                type: "GET",
                dataType: "json",
                success: function (data) {

                    let outputhtml = "";

                    //Foreach banlist, add the cards to the appropriate div with the appropriate class for text or image view
                    jQuery.each(data, function (i, item) {

                        if(curview == 'image') {
                            var tView = 'd-none';
                            var iView = 'd-block';
                            var cview = '';
                        }else{
                            var tView = 'd-flex';
                            var iView = 'd-none';
                            var cview = 'd-block w-100';
                        }

                        outputhtml += '<span class="card-row '+cview+'" data-status="'+item.status_text+'">';

                        //Text View
                        outputhtml += '<span class="shadow-sm border p-2 m-2 card-list-text '+tView+' justify-content-between">';
                        outputhtml +=  '<span style="width:200px;" class="text-truncate"><a href="/card/'+item.pretty_url+'" target="_blank"><strong>'+item.name+'</strong></a></span>';
                        outputhtml +=  '<span class="card-type">'+item.type+'</span>';
                        outputhtml +=  '<span><strong>'+item.status_text+'</strong></span>';
                        outputhtml += '</span>';

                        //Image View
                        outputhtml += '<span class="m-2 card-list-img text-container-icon '+iView+'">';
                        outputhtml += '<figure><a href="/card/'+item.pretty_url+'" target="_blank"><img data-src="https://images.ygoprodeck.com/images/cards_small/'+item.id+'.jpg" class="lazy card-img" alt="'+item.name+'" loading="lazy">';
                        outputhtml += '</a><figcaption class="text-center ml-auto mr-auto" style="width:150px;"><span class="d-block card-name text-truncate small mt-1 font-weight-bold">'+item.name+'</span><span class="d-block card-name text-truncate small mt-1">'+item.status_text+'</span></figcaption>';
                        outputhtml += '</figure>';
                        outputhtml += '</span>';


                        outputhtml += '</span>';

                    });

                    //Loop through outputhtml and add it to the appropriate div by data-status
                    jQuery("#forbidden").html("").html('<h3 id="forbidden-header" class="d-block w-100 text-center">Forbidden</h3>');
                    jQuery("#limited").html("").html('<h3 id="limited-header" class="d-block w-100 text-center">Limited</h3>');
                    jQuery("#semilimited").html("").html('<h3 id="semilimited-header" class="d-block w-100 text-center">Semi-Limited</h3>');
                    jQuery(outputhtml).each(function(){
                        if(jQuery(this).attr('data-status') == 'Forbidden'){
                            jQuery("#forbidden").append(jQuery(this));
                        }
                        if(jQuery(this).attr('data-status') == 'Limited'){
                            jQuery("#limited").append(jQuery(this));
                        }
                        if(jQuery(this).attr('data-status') == 'Semi-Limited'){
                            jQuery("#semilimited").append(jQuery(this));
                        }
                    });

                    //Lazy load images
                    jQuery('.lazy').Lazy({
                        // your configuration goes here
                        defaultImage: "https://images.ygoprodeck.com/images/assets/CardBack.jpg",
                        scrollDirection: 'vertical',
                        effect: "fadeIn",
                        effectTime: 500,
                        threshold: 0,
                        visibleOnly: true,
                        onError: function(element) {
                            console.log('error loading ' + element.data('src'));
                        }
                    });


                },
                error: function () {
                    //If the banlist is not valid, display an error
                    jQuery("#banlist").html("<p style='text-align:center'>There was an error getting the banlist. Please try again later.</p>");
                }
            });
        }

        //Toggle View Function
        function changeView(view){
            curview = view;
            if(view == 'text'){
                jQuery('.card-list-img').removeClass('d-block').addClass('d-none');
                jQuery('.card-list-text').removeClass('d-none').addClass('d-flex');
                jQuery('.card-row').addClass('d-block w-100');

                //Buttons
                jQuery('#imageButton').removeClass('btn-primary').addClass('btn-secondary');
                jQuery('#textButton').removeClass('btn-secondary').addClass('btn-primary');
            }else{
                jQuery('.card-list-img').removeClass('d-none').addClass('d-block');
                jQuery('.card-list-text').removeClass('d-flex').addClass('d-none');
                jQuery('.card-row').removeClass('d-block w-100');

                //Buttons
                jQuery('#textButton').removeClass('btn-primary').addClass('btn-secondary');
                jQuery('#imageButton').removeClass('btn-secondary').addClass('btn-primary');
            }
        }
    </script>
